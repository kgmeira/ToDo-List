<script>

/* ===== ESTADO ===== */
let tasks = JSON.parse(localStorage.getItem("tasks")) || [];
let score = localStorage.getItem("score") || 0;
let filter = "all";

/* ===== SALVAR ===== */
function save() {
    localStorage.setItem("tasks", JSON.stringify(tasks));
    localStorage.setItem("score", score);
}

/* ===== FILTRO ===== */
function setFilter(f) {
    filter = f;
    render();
}

/* ===== DATA ===== */
function isToday(date) {
    return date === new Date().toISOString().split("T")[0];
}

function isWeek(date) {
    const now = new Date();
    const taskDate = new Date(date);
    const diff = (taskDate - now) / (1000 * 60 * 60 * 24);
    return diff <= 7 && diff >= 0;
}

/* ===== NOTIFICAÇÃO ===== */
function notifyTodayTasks() {
    const todayTasks = tasks.filter(t => isToday(t.date) && !t.completed);

    if (todayTasks.length > 0) {
        alert("🔔 Você tem " + todayTasks.length + " tarefa(s) hoje!");
    }
}

/* ===== PROGRESSO ===== */
function updateProgress() {
    const done = tasks.filter(t => t.completed).length;
    const total = tasks.length;

    const percent = total ? (done / total) * 100 : 0;

    document.getElementById("progressBar").style.width = percent + "%";
}

/* ===== RENDER ===== */
function render() {
    const list = document.getElementById("taskList");
    list.innerHTML = "";

    let filtered = tasks;

    if (filter === "today") {
        filtered = tasks.filter(t => isToday(t.date));
    }

    if (filter === "week") {
        filtered = tasks.filter(t => isWeek(t.date));
    }

    filtered.forEach((task, i) => {

        const li = document.createElement("li");
        li.className = task.priority;

        const checkbox = document.createElement("input");
        checkbox.type = "checkbox";
        checkbox.checked = task.completed;

        checkbox.onchange = () => {
            task.completed = checkbox.checked;

            if (checkbox.checked) {
                score++;
            } else {
                score--;
            }

            save();
            render();
        };

        const span = document.createElement("span");
        span.textContent = task.text;

        if (task.completed) span.classList.add("completed");

        const remove = document.createElement("button");
        remove.textContent = "X";

        remove.onclick = () => {
            tasks.splice(i, 1);
            save();
            render();
        };

        const date = document.createElement("div");
        date.textContent = "📅 " + (task.date || "Sem data");

        li.appendChild(checkbox);
        li.appendChild(span);
        li.appendChild(remove);
        li.appendChild(date);

        list.appendChild(li);
    });

    updateProgress();
    document.getElementById("score").textContent = score;
}

/* ===== ADD ===== */
function addTask() {
    const text = document.getElementById("taskInput").value;
    const priority = document.getElementById("priority").value;
    const date = document.getElementById("date").value;

    if (!text.trim()) return;

    tasks.push({ text, priority, date, completed: false });

    save();
    render();
}

/* ===== DARK MODE ===== */
function toggleDark() {
    document.body.classList.toggle("dark");
}

/* ===== INIT ===== */
render();
notifyTodayTasks();

⚡ 2. COMO PODERIA SER MELHOR / OTIMIZADO

Agora vem a parte importante 👇

🔥 1. Problema: render recria tudo sempre
list.innerHTML = "";

❌ Ruim porque:

Recria toda a DOM a cada ação
Ineficiente com muitas tarefas

✅ Melhor:

Atualizar só o item alterado
Ou usar documentFragment
🔥 2. Problema: índice como ID
tasks.splice(i, 1);

❌ Ruim porque:

Índice muda quando remove itens
Pode causar bugs

✅ Melhor:

id: Date.now()
🔥 3. Problema: lógica + UI misturados

Tudo está dentro de render()

❌ Ruim porque:

Código difícil de manter
Baixa escalabilidade

✅ Melhor separar:

lógica (dados)
renderização (UI)
🔥 4. Problema: score pode ficar negativo
score--;

❌ Bug possível

✅ Melhor:

score = Math.max(0, score - 1);
🔥 5. Problema: notificação fraca
alert(...)

❌ UX ruim

✅ Melhor:

usar Notification API
ou toast customizado
🔥 6. Problema: dark mode não salva

❌ Ao recarregar, perde

✅ Melhor:

localStorage.setItem("dark", true)
🔥 7. Problema: sem validação de data

❌ Pode salvar tarefa sem data

✅ Melhor:

validar ou definir padrão
🔥 8. Problema: acessibilidade inexistente

❌ Sem:

labels
aria
foco

👉 Isso derruba qualidade profissional
